\tikzset{
  pics/dotted grid/.style={
    code={
      \foreach \i in {-18,-17,...,18}{
        \foreach \j in {0,1,2,...,22}{
          \draw[draw=white, fill=lightgray, opacity=.5] 
            (0,-1cm) ++(\i * 4mm, -\j * 4mm) circle (1pt);
        }
      }
    }
  }
}

      {\includegraphics[width={\slideswidth}, page=\slide]{\slides}};
    \pic at (current page.center) {dotted grid};
  \end{tikzpicture}

}
